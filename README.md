{
    void Start()

    {
        for (int i = 1; i <= 20; i++)
        {
            if(i % 2 == 0) 
                // i % 2 == 0은
                // “i를 2로 나눈 나머지가 0이면”
                //즉, **i가 2의 배수(짝수)**라는 뜻이에요.
            {
                Debug.Log(i + "는 짝수입니다!");
            }
            else
            {
                Debug.Log(i + "는 홀수입니다!");
            }
        }
    }   
}
